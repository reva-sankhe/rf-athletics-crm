function getTier(mas) {
  if (!mas || mas <= 0) return null;
  return tiers.find(t => mas >= t.masMin && mas < t.masMax) || tiers[tiers.length-1];
}

function fmtB(v) {
  if (v == null) return '—';
  const m = Math.floor(v);
  const s = Math.round((v - m) * 60);
  return m + ':' + (s < 10 ? '0' : '') + s;
}

function getJ(code) { return janData[code] || null; }
function getM(code) { return marData[code] || null; }

// ── METRICS ──────────────────────────────────────────────────────────────────
function buildMetrics() {
  const sharksWithMar = master.filter(m => getM(m.code)?.bronco);
  const sharksWithJan = master.filter(m => getJ(m.code)?.bronco);
  const both = master.filter(m => getJ(m.code)?.bronco && getM(m.code)?.bronco);
  const improved = both.filter(m => getM(m.code).bronco < getJ(m.code).bronco);
  const avgJan = sharksWithJan.reduce((s,m) => s + getJ(m.code).bronco, 0) / sharksWithJan.length;
  const avgMar = sharksWithMar.reduce((s,m) => s + getM(m.code).bronco, 0) / sharksWithMar.length;

  document.getElementById('metricStrip').innerHTML = `
    <div class="metric"><div class="metric-label">Squad (Sharks)</div><div class="metric-val">${master.length}</div><div class="metric-foot">registered players</div></div>
    <div class="metric"><div class="metric-label">Tested both</div><div class="metric-val accent">${both.length}</div><div class="metric-foot">Jan + Mar</div></div>
    <div class="metric"><div class="metric-label">Avg bronco Jan</div><div class="metric-val">${fmtB(avgJan)}</div><div class="metric-foot">team mean</div></div>
    <div class="metric"><div class="metric-label">Avg bronco Mar</div><div class="metric-val">${fmtB(avgMar)}</div><div class="metric-foot">team mean</div></div>
    <div class="metric"><div class="metric-label">Improved</div><div class="metric-val accent">${improved.length}</div><div class="metric-foot">of ${both.length} comparable</div></div>
    <div class="metric"><div class="metric-label">Declined</div><div class="metric-val red">${both.length - improved.length}</div><div class="metric-foot">of ${both.length} comparable</div></div>
  `;
}

// ── BENCHMARK ────────────────────────────────────────────────────────────────
function buildBenchmark() {
  const marSharks = master.filter(m => getM(m.code)?.mas && getM(m.code).mas > 0);

  const tierRows = tiers.map(t => {
    const players = marSharks.filter(m => getM(m.code).mas >= t.masMin && getM(m.code).mas < t.masMax);
    const dots = players.map(p => `<div class="dot" style="background:${t.color}22; color:${t.color}; border:1px solid ${t.color}44;" title="${p.name} (${getM(p.code).mas.toFixed(2)} m/s)">${p.name.slice(0,2).toUpperCase()}<div class="dot-tip">${p.name} · ${getM(p.code).mas.toFixed(2)} m/s</div></div>`).join('');
    return `<div class="bench-row" style="background:${t.color}0a; border:1px solid ${t.color}22;">
      <div class="bench-label" style="color:${t.color};">${t.label}</div>
      <div class="bench-range">${t.range}</div>
      <div class="bench-desc">${t.desc}</div>
      <div class="bench-count" style="color:${t.color};">${players.length}</div>
    </div>
    ${dots ? `<div class="player-dots" style="padding:0 10px 8px;">${dots}</div>` : ''}`;
  }).join('');

  // Build chart showing all players vs tiers
  const sorted = [...marSharks].sort((a,b) => getM(b.code).mas - getM(a.code).mas);

  document.getElementById('tab-benchmark').innerHTML = `
    <div class="card">
      <div class="card-title">Player MAS vs female athlete benchmarks (Mar 2026)</div>
      <div class="chart-wrap" style="height:${sorted.length*26+80}px;"><canvas id="benchChart"></canvas></div>
    </div>
    <div class="card">
      <div class="card-title">Tier distribution — Sharks squad (Mar 2026)</div>
      <div class="bench-strip">${tierRows}</div>
    </div>`;

  const barColors = sorted.map(m => getTier(getM(m.code).mas)?.color || '#6b7280');

  const ctx = document.getElementById('benchChart').getContext('2d');

  // Draw tier background zones as annotation-style horizontal bands using plugin
  new Chart(ctx, {
    type: 'bar',
    data: {
      labels: sorted.map(m => m.name),
      datasets: [{
        label: 'MAS (m/s)',
        data: sorted.map(m => parseFloat(getM(m.code).mas.toFixed(3))),
        backgroundColor: barColors,
        borderRadius: 3,
      }]
    },
    options: {
      indexAxis: 'y',
      responsive: true,
      maintainAspectRatio: false,
      plugins: {
        legend: { display: false },
        tooltip: {
          callbacks: {
            label: ctx => {
              const tier = getTier(ctx.raw);
              return `${ctx.raw.toFixed(2)} m/s — ${tier ? tier.label : 'Unknown'}`;
            }
          }
        }
      },
      scales: {
        x: {
          min: 2.4, max: 4.8,
          grid: { color: 'rgba(255,255,255,0.05)' },
          ticks: { color: '#5a5956', font: { size: 10 } },
          border: { color: 'rgba(255,255,255,0.08)' }
        },
        y: { ticks: { color: '#9a9892', font: { size: 11 } }, grid: { display: false }, border: { color: 'rgba(255,255,255,0.08)' } }
      }
    },
    plugins: [{
      afterDraw(chart) {
        const {ctx, chartArea, scales} = chart;
        tiers.forEach(t => {
          if (t.masMax === 99) return;
          const x = scales.x.getPixelForValue(t.masMin);
          ctx.save();
          ctx.strokeStyle = t.color + '55';
          ctx.lineWidth = 1;
          ctx.setLineDash([4,3]);
          ctx.beginPath();
          ctx.moveTo(x, chartArea.top);
          ctx.lineTo(x, chartArea.bottom);
          ctx.stroke();
          ctx.fillStyle = t.color + 'aa';
          ctx.font = '9px DM Mono, monospace';
          ctx.fillText(t.label, x + 3, chartArea.top + 10);
          ctx.restore();
        });
      }
    }]
  });
}

// ── PLAYERS ───────────────────────────────────────────────────────────────────
function buildPlayers() {
  let fPos = 'all', fRange = 'all';

  function render() {
    const rows = master.filter(m => {
      if (fPos !== 'all' && m.pos !== fPos) return false;
      if (fRange !== 'all' && m.range !== fRange) return false;
      return true;
    });

    const tbody = rows.map(m => {
      const j = getJ(m.code), mr = getM(m.code);
      const jb = j?.bronco, mb = mr?.bronco;
      let badge = '';
      if (jb && mb) {
        const d = jb - mb;
        if (d > 0.05) badge = `<span class="badge badge-green">▲ ${d.toFixed(2)}</span>`;
        else if (d < -0.05) badge = `<span class="badge badge-red">▼ ${Math.abs(d).toFixed(2)}</span>`;
        else badge = `<span class="badge badge-amber">—</span>`;
      } else badge = `<span class="badge badge-gray">no compare</span>`;

      const tier = getTier(mr?.mas);
      const tierBadge = tier ? `<span style="font-size:10px; color:${tier.color};">${tier.label}</span>` : '—';

      return `<tr>
        <td><span class="pos pos-${m.pos||'GK'}">${m.pos||'?'}</span></td>
        <td style="font-weight:500;">${m.name}</td>
        <td><span class="badge badge-gray">${m.range||'—'}</span></td>
        <td style="font-family:'DM Mono',monospace;">${jb ? fmtB(jb) : '<span style="color:#5a5956;">—</span>'}</td>
        <td style="font-family:'DM Mono',monospace;">${mb ? fmtB(mb) : '<span style="color:#5a5956;">—</span>'}</td>
        <td style="font-family:'DM Mono',monospace; font-size:12px;">${mr?.mas > 0 ? mr.mas.toFixed(2) : '—'}</td>
        <td>${tierBadge}</td>
        <td>${badge}</td>
      </tr>`;
    }).join('');

    document.getElementById('player-tbody').innerHTML = tbody;
  }

  document.getElementById('tab-players').innerHTML = `
    <div class="filter-row">
      <span class="filter-label">Position</span>
      <select id="fpos" onchange="updatePlayerFilter()">
        <option value="all">All</option><option>F</option><option>M</option><option>D</option><option>GK</option>
      </select>
      <span class="filter-label">Age group</span>
      <select id="frange" onchange="updatePlayerFilter()">
        <option value="all">All</option><option>U18</option><option>18-24</option><option>25+</option>
      </select>
    </div>
    <div class="card" style="padding:0; overflow:hidden;">
      <div class="tbl-wrap">
        <table>
          <thead><tr>
            <th></th><th>Player</th><th>Group</th>
            <th>Jan bronco</th><th>Mar bronco</th>
            <th>MAS (m/s)</th><th>Tier</th><th>Change</th>
          </tr></thead>
          <tbody id="player-tbody"></tbody>
        </table>
      </div>
    </div>`;

  window.updatePlayerFilter = () => {
    fPos = document.getElementById('fpos').value;
    fRange = document.getElementById('frange').value;
    render();
  };
  render();
}

// ── AGE ────────────────────────────────────────────────────────────────────────
function buildAge() {
  const ranges = [
    {key:'U18',label:'Under 18',color:'#ff5a5a'},
    {key:'18-24',label:'18 – 24',color:'#5a9cf5'},
    {key:'25+',label:'25 & over',color:'#b5f23a'},
  ];

  const cards = ranges.map(r => {
    const inR = master.filter(m => m.range === r.key);
    const janR = inR.filter(m => getJ(m.code)?.bronco);
    const marR = inR.filter(m => getM(m.code)?.bronco);
    const both = inR.filter(m => getJ(m.code)?.bronco && getM(m.code)?.bronco);
    const avgJ = janR.length ? janR.reduce((s,m)=>s+getJ(m.code).bronco,0)/janR.length : null;
    const avgM = marR.length ? marR.reduce((s,m)=>s+getM(m.code).bronco,0)/marR.length : null;
    const imp = both.filter(m => getM(m.code).bronco < getJ(m.code).bronco).length;

    const bars = [...marR].sort((a,b)=>getM(a.code).bronco-getM(b.code).bronco).map(m => {

  document.getElementById('tab-age').innerHTML = `
